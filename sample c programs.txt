9.//program to find a palindrome of a number

void main()
{
   int n,r,sum=0,temp;
   printf("enter the number:");
   scanf("%d",&n);
   temp=n;
   while(n!=0)
   {
       r=n%10;
       sum=sum*10+r;
       n=n/10;
   }
   n=temp;
   if(sum==n)

       printf("the given number %d is a palindrome",n);

   }
}
